# Let's create a function to clean our text data
def clean_text(text):
    """
    Clean and preprocess text for analysis
    """
    # Handle missing values
    if pd.isna(text):
        return ""
    
    # Convert to string and lowercase
    text = str(text).lower()
    
    # Remove special characters and numbers
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    
    # Remove extra whitespace
    text = ' '.join(text.split())
    
    return text

# Now let's do advanced preprocessing with tokenization and lemmatization
def preprocess_text_advanced(text):
    """
    Advanced preprocessing with tokenization and lemmatization
    """
    if not text or len(text.strip()) == 0:
        return ""
    
    try:
        # Tokenize the text
        tokens = word_tokenize(text)
        
        # Remove stopwords
        stop_words = set(stopwords.words('english'))
        # Add some financial-specific stop words
        financial_stopwords = {'said', 'says', 'company', 'stock', 'share', 'market'}
        stop_words.update(financial_stopwords)
        
        # Filter tokens
        tokens = [token for token in tokens 
                 if token not in stop_words and len(token) > 2]
        
        # Lemmatize tokens
        lemmatizer = WordNetLemmatizer()
        tokens = [lemmatizer.lemmatize(token) for token in tokens]
        
        return ' '.join(tokens)
        
    except Exception as e:
        print(f"Error processing text: {e}")
        return text  # Return original if processing fails

# Create a function to analyze sentiment
def analyze_sentiment(text):
    """
    Analyze sentiment using TextBlob
    Returns polarity, subjectivity, and sentiment label
    """
    if not text or len(text.strip()) == 0:
        return 0.0, 0.0, 'Neutral'
    
    try:
        # Create TextBlob object
        blob = TextBlob(text)
        
        # Get polarity and subjectivity
        polarity = blob.sentiment.polarity
        subjectivity = blob.sentiment.subjectivity
        
        # Classify sentiment based on polarity
        if polarity > 0.1:
            sentiment = 'Positive'
        elif polarity < -0.1:
            sentiment = 'Negative'
        else:
            sentiment = 'Neutral'
        
        return polarity, subjectivity, sentiment
    
    except Exception as e:
        print(f"Error analyzing sentiment: {e}")
        return 0.0, 0.0, 'Neutral'

# First, let's find the optimal number of clusters
print("Finding optimal number of clusters...")

silhouette_scores = []
K_range = range(2, 8)  # Try 2 to 7 clusters

for k in K_range:
    # Create K-means model
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    
    # Fit the model and get cluster labels
    cluster_labels = kmeans.fit_predict(document_term_matrix)
    
    # Calculate silhouette score
    score = silhouette_score(document_term_matrix, cluster_labels)
    silhouette_scores.append(score)
    
    print(f"   k={k} clusters: Silhouette Score = {score:.3f}")

# Find the best number of clusters
optimal_k = K_range[np.argmax(silhouette_scores)]
best_score = max(silhouette_scores)

print(f"\n Optimal number of clusters: {optimal_k}")
